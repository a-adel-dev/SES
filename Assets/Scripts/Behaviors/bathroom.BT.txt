tree("OriginalRoot")
	sequence
		random(.6, .4)
			tree("VisitLocker")
			tree("VisitBathroom")
		ClearStatus
		WaitRandom(5.0, 10.0)


tree("VisitBathroom")
		Fallback
			while not IsBusy
				Fallback
					sequence
						while ClearToGo
							sequence
								ExitClass
								GetBathroom
								GoToBathroom
						while WentToBathroom
							sequence
								ConfirmBathroomReach
								GetToilet
								GoToToilet
								ConfirmReach
								WaitRandom(5.0, 10.0)
								ExitBathroom
								BackToDesk
								ConfirmReach
								ReassignOriginalClassroom
								ClearVisitStatus
						while not WentToBathroom
							sequence
								BackToDesk
								ConfirmReach
								ResetClassRoom
								ClearStatus
		

tree("VisitLocker")
	while ClearToGo
		sequence
			GoToLocker
			ConfirmReach
			WaitRandom(4.0, 8.0)
			ReleaseLocker
			BackToDesk
			ConfirmReach
			ClearStatus
			Wait(5.0)
