tree("Root")
	sequence
		mute
			random(.6, .4)
				tree("VisitLocker")
				tree("VisitBathroom")
		sequence
			ClearVisitStatus
			WaitRandom(5.0, 10.0)


tree("VisitBathroom")
	while ClearToGo
		sequence
			mute 
				sequence			
					SetDoingBehavior(true)
					ExitClass
					GetBathroom
					GoToBathroom
					ConfirmBathroomReach
					GetToilet
			while GotSpot
				sequence
					GoToToilet
					ConfirmReach
					WaitRandom(5.0, 10.0)
			sequence
				ExitBathroom
				BackToDesk
				ConfirmReach
				ReassignOriginalClassroom
				SetDoingBehavior(false)
						
	

tree("VisitLocker")
	while ClearToGo
		sequence
			SetDoingBehavior(true)
			GoToLocker
			ConfirmReach
			WaitRandom(4.0, 8.0)
			ReleaseLocker
			BackToDesk
			ConfirmReach
			SetDoingBehavior(false)
